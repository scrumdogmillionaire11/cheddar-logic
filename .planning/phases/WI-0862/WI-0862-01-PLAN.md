In `parseSnapshotPayload` (around line 325), replace the markets extraction line:

```js
// BEFORE:
markets: payload && payload.markets ? payload.markets : {},

// AFTER:
markets: payload?.markets
  ?? (payload?.spreads || payload?.totals || payload?.h2h ? payload : {}),
```

This handles two shapes without breaking existing tests:
- **Stored shape** (prod): `{ spreads: [...], totals: [...], h2h: [...] }` → falls back to `payload` itself
- **Wrapped shape** (unit tests): `{ markets: { spreads, totals, h2h } }` → uses `payload.markets` as before
- **Empty/null payload**: falls through to `{}`

Do not touch any other logic. This is a 3-line change (remove old line, add new lines).

  it('scanLineDiscrepancies finds line gap when raw_data is flat { spreads, totals, h2h } (prod shape)', () => {

    const snapshot = {
      game_id: 'test-prod-shape-001',
      sport: 'baseball_mlb',
      captured_at: new Date().toISOString(),
      raw_data: JSON.stringify({
        // flat — no "markets:" wrapper, this is exactly what pull_odds_hourly stores
        spreads: [
          { bookmaker: 'draftkings', homeSpread: -1.5, awaySpread: 1.5 },
          { bookmaker: 'fanduel',    homeSpread: -1.5, awaySpread: 1.5 },
          { bookmaker: 'betmgm',     homeSpread: -3.0, awaySpread: 3.0 }, // outlier
        ],
        totals: [],
        h2h: [],
      }),
    };

    const gaps = scanLineDiscrepancies([snapshot], { minBooks: 3, recencyWindowMs: 60 * 60 * 1000 });
    // Should detect the 1.5pt gap (draftkings/fanduel vs betmgm)
    assert.ok(gaps.length > 0, 'Expected at least 1 LineGap from prod-shaped snapshot');
    assert.strictEqual(gaps[0].sport, 'baseball_mlb');
  });

  it('scanLineDiscrepancies returns [] for wrapped shape { markets: { spreads, totals, h2h } } (backward compat)', () => {

      game_id: 'test-wrapped-shape-001',

      captured_at: new Date().toISOString(),
      raw_data: JSON.stringify({

    const gaps = scanLineDiscrepancies([snapshot], { minBooks: 3, recencyWindowMs: 60 * 60 * 1000 });
    assert.ok(gaps.length > 0, 'Expected at least 1 LineGap from wrapped-shape snapshot');
  });
});
