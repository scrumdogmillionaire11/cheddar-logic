```typescript
import { getOddsSnapshots } from '@cheddar-logic/data';

// Module-level cache — survives across requests in the same Node process
const CACHE_TTL_MS = 4.5 * 60 * 1000;  // 4 min 30 sec
const LOOKBACK_MS  = 30 * 60 * 1000;   // 30-minute snapshot window

interface LineGap {
  gameId: string;
  sport: string;
  outlierBook: string;
  delta: number;
  tier: 'TRIGGER' | 'WATCH';
  [key: string]: unknown;
}

interface OddsGap {
  gameId: string;
  sport: string;
  impliedEdgePct: number;
  tier: 'TRIGGER' | 'WATCH';
  [key: string]: unknown;
}

interface MarketPulseResponse {
  scannedAt: string;
  lineGaps: LineGap[];
  oddsGaps: OddsGap[];
  meta: {
    gamesScanned: number;
    booksScanned: number;
    lineGapCount: number;
    oddsGapCount: number;
  };
}

interface CacheEntry {
  payload: MarketPulseResponse;
  expiresAt: number;
}

const cache = new Map<string, CacheEntry>();

const VALID_SPORTS = ['ALL', 'NBA', 'MLB', 'NHL'];

function extractUniqueBooks(snapshots: { raw_data: string }[]): Set<string> {
  const books = new Set<string>();
  for (const snap of snapshots) {
    try {
      const parsed = JSON.parse(snap.raw_data ?? '{}');
      const markets = parsed?.markets ?? parsed;
      const allEntries = [
        ...(markets?.spreads ?? []),
        ...(markets?.totals ?? []),
        ...(markets?.h2h ?? []),
      ];
      for (const entry of allEntries) {
        if (entry?.book) books.add(entry.book);
      }
    } catch {
      // malformed raw_data — skip
    }
  }
  return books;
}

function serveResponse(payload: MarketPulseResponse, includeWatch: boolean): Response {
  if (includeWatch) {
    return Response.json(payload);
  }
  return Response.json({
    ...payload,
    lineGaps: payload.lineGaps.filter((g) => g.tier === 'TRIGGER'),
    oddsGaps: payload.oddsGaps.filter((g) => g.tier === 'TRIGGER'),
  });
}

export async function GET(request: Request): Promise<Response> {
  const { searchParams } = new URL(request.url);
  const sport        = (searchParams.get('sport') ?? 'ALL').toUpperCase();
  const includeWatch = searchParams.get('includeWatch') === 'true';

  if (!VALID_SPORTS.includes(sport)) {
    return Response.json({ error: 'Invalid sport' }, { status: 400 });
  }

  // Serve from cache if still fresh
  const cached = cache.get(sport);
  if (cached && cached.expiresAt > Date.now()) {
    return serveResponse(cached.payload, includeWatch);
  }

  // Lazy-import scanner (CJS package; dynamic import avoids Edge/Node conflict)
  const { scanLineDiscrepancies, scanOddsDiscrepancies } =
    await import('@cheddar-logic/models');

  const sinceUtc = new Date(Date.now() - LOOKBACK_MS).toISOString();

  const snapshots =
    sport === 'ALL'
      ? [
          ...getOddsSnapshots('NBA', sinceUtc),
          ...getOddsSnapshots('MLB', sinceUtc),
          ...getOddsSnapshots('NHL', sinceUtc),
        ]
      : getOddsSnapshots(sport, sinceUtc);

  // Pass 1 — line discrepancies across all snapshots
  const allLineGaps: LineGap[] = (scanLineDiscrepancies(snapshots) as LineGap[])
    .sort((a, b) => b.delta - a.delta);

  // Pass 2 — odds discrepancies on games with no line gap
  const lineGapGameIds = new Set(allLineGaps.map((g) => g.gameId));
  const cleanSnapshots = (snapshots as { game_id: string }[]).filter(
    (s) => !lineGapGameIds.has(s.game_id),
  );
  const allOddsGaps: OddsGap[] = (
    scanOddsDiscrepancies(cleanSnapshots) as OddsGap[]
  ).sort((a, b) => b.impliedEdgePct - a.impliedEdgePct);

  const uniqueGames = new Set(
    (snapshots as { game_id: string }[]).map((s) => s.game_id),
  );
  const uniqueBooks = extractUniqueBooks(snapshots as { raw_data: string }[]);

  const payload: MarketPulseResponse = {
    scannedAt: new Date().toISOString(),
    lineGaps: allLineGaps,
    oddsGaps: allOddsGaps,
    meta: {
      gamesScanned: uniqueGames.size,
      booksScanned: uniqueBooks.size,
      lineGapCount: allLineGaps.length,
      oddsGapCount: allOddsGaps.length,
    },
  };

  cache.set(sport, { payload, expiresAt: Date.now() + CACHE_TTL_MS });
  return serveResponse(payload, includeWatch);
}
```
