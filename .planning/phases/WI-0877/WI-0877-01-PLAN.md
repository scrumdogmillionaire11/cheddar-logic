The function signature and contract:

```js
/**
 * Compute a full-model F5 total driver when no market line is available.
 *
 * Uses projectTeamF5RunsAgainstStarter() for both sides against a synthetic
 * market line (3.5 or 4.5 — the two standard F5 totals). Returns null when
 * offense profiles are absent or either projection degrades to f5_runs: null;
 * caller falls back to SYNTHETIC_FALLBACK PASS behavior.
 *
 * @param {object} mlb  parsed raw_data.mlb object
 * @param {object} context  { park_run_factor, temp_f, wind_mph, wind_dir, roof }
 * @param {string} gameId
 * @returns {object|null}
 */
function computeSyntheticLineF5Driver(mlb, context, gameId) {
  if (!mlb.home_offense_profile || !mlb.away_offense_profile) return null;

  // away offense faces home starter; home offense faces away starter
  const homeResult = projectTeamF5RunsAgainstStarter(
    mlb.home_pitcher ?? null,
    mlb.away_offense_profile,
    context,
  );
  const awayResult = projectTeamF5RunsAgainstStarter(
    mlb.away_pitcher ?? null,
    mlb.home_offense_profile,
    context,
  );

  if (homeResult.f5_runs === null || awayResult.f5_runs === null) return null;

  const projectedBase = homeResult.f5_runs + awayResult.f5_runs;
  // Standard F5 lines: if projection < 4.0 the market likely posts 3.5; else 4.5
  const syntheticLine = projectedBase < 4.0 ? 3.5 : 4.5;
  const edge = projectedBase - syntheticLine;

  // Confidence: start at 6 (synthetic line risk vs 7 for real-odds path).
  // Subtract 1 per side with degraded_inputs (missing park/weather/etc.). Floor 4.
  let confidence = 6;
  if ((homeResult.degraded_inputs ?? []).length > 0) confidence -= 1;
  if ((awayResult.degraded_inputs ?? []).length > 0) confidence -= 1;
  confidence = Math.max(4, confidence);

  const isOver = edge >= MLB_F5_EDGE_THRESHOLD && confidence >= 8;
  const isUnder = edge <= -MLB_F5_EDGE_THRESHOLD && confidence >= 8;
  const evThresholdPassed = isOver || isUnder;
  const prediction = isOver ? 'OVER' : isUnder ? 'UNDER' : (edge >= 0 ? 'OVER' : 'UNDER');
  const status = evThresholdPassed ? 'FIRE' : (Math.abs(edge) >= MLB_F5_EDGE_THRESHOLD ? 'WATCH' : 'PASS');
  const action = evThresholdPassed ? 'FIRE' : 'PASS';
  const classification = status;

  const reasonCodes = ['SYNTHETIC_LINE_ASSUMPTION'];
  if ((homeResult.degraded_inputs ?? []).length > 0) reasonCodes.push('DEGRADED_INPUT_HOME');
  if ((awayResult.degraded_inputs ?? []).length > 0) reasonCodes.push('DEGRADED_INPUT_AWAY');
  if (!evThresholdPassed) reasonCodes.push('PASS_NO_EDGE');

  return {
    market: 'f5_total',
    prediction,
    confidence: confidence / 10,
    status,
    action,
    classification,
    ev_threshold_passed: evThresholdPassed,
    projection_source: 'FULL_MODEL',
    status_cap: status,
    reason_codes: reasonCodes,
    missing_inputs: [],
    playability: {
      over_playable_at_or_below: syntheticLine - MLB_F5_EDGE_THRESHOLD,
      under_playable_at_or_above: syntheticLine + MLB_F5_EDGE_THRESHOLD,
    },
    projection: {
      projected_total: projectedBase,
      projected_total_low: Math.max(0, projectedBase - 0.5),
      projected_total_high: projectedBase + 0.5,
      projected_home_f5_runs: Math.round(homeResult.f5_runs * 10) / 10,
      projected_away_f5_runs: Math.round(awayResult.f5_runs * 10) / 10,
    },
    reasoning: `F5 FULL_MODEL synthetic-line: projected ${projectedBase.toFixed(2)} vs assumed line ${syntheticLine} (edge ${edge >= 0 ? '+' : ''}${edge.toFixed(2)}, conf ${confidence}/10, ${status}); synthetic line assumed [-115/-115 vig], reason: SYNTHETIC_LINE_ASSUMPTION`,
    drivers: [{
      type: 'mlb-f5-synthetic-edge',
      edge: parseFloat(edge.toFixed(3)),
      projected_total: projectedBase,
      synthetic_line: syntheticLine,
      projection_source: 'FULL_MODEL',
      home_f5_runs: homeResult.f5_runs,
      away_f5_runs: awayResult.f5_runs,
    }],
    without_odds_mode: true,
    synthetic_line: syntheticLine,
    synthetic_line_assumption: true,
  };
}
```
