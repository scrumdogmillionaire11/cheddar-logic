```python
# Load production edge sample
edge_sample = pd.read_json("risk_data/edge_sample.json")

# Required contract gate
required_cols = ["calibrated_fair_prob", "implied_prob", "outcome", "resolved_at"]
missing_cols = [c for c in required_cols if c not in edge_sample.columns]
assert not missing_cols, f"Missing required columns: {missing_cols}"
assert edge_sample[required_cols].notnull().all().all(), "Nulls in required columns"
assert ((edge_sample["calibrated_fair_prob"] >= 0) & (edge_sample["calibrated_fair_prob"] <= 1)).all()
assert ((edge_sample["implied_prob"] >= 0) & (edge_sample["implied_prob"] <= 1)).all()
assert edge_sample["outcome"].isin([0, 1]).all()

edge_sample["net_edge_pp"] = (edge_sample["calibrated_fair_prob"] - edge_sample["implied_prob"]) * 100

plt.hist(edge_sample["net_edge_pp"], bins=40)
plt.axvline(0, color="red", ls="--")
plt.title("Empirical Net Edge Distribution (post-calibration)")
```
