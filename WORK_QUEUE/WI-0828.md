---

## Background

The statistical audit was manual. It will drift. The two fixes here make it permanent:

**Ablation tests** verify that removing a feature actually changes the projection — if it doesn't, the feature is either redundant or already bypassed. These run in CI alongside unit tests.

**Standardized output** ensures that cross-sport monitoring, the execution gate (WI-0824), and the calibration layer (WI-0825) all read from a predictable shape regardless of sport. Currently each model returns a different object structure, which forces every consumer to handle sport-specific field names.

## Standard Model Output Shape

```js
/**
 * Canonical model output shape.
 * Every model function must return this or a superset of this.
 *
 * @typedef {object} ModelOutput
 * @property {string}  market           - e.g. 'MLB_F5_TOTAL', 'NHL_TOTAL', 'NBA_TOTAL'
 * @property {'MODEL_OK'|'DEGRADED'|'NO_BET'} status
 * @property {number|null}  fairProb    - model fair probability (0–1); null if NO_BET
 * @property {number|null}  fairLine    - model fair line (totals/spreads); null for ML
 * @property {number|null}  confidence  - 0–1 confidence; 0 if NO_BET
 * @property {object}       featuresUsed
 * @property {string[]}     missingOptional
 * @property {string[]}     missingCritical
 * @property {object}       diagnostics
 * @property {boolean}      diagnostics.leakageSafe
 * @property {string}       diagnostics.version
 */

function buildModelOutput({
  market,
  status,
  fairProb = null,
  fairLine = null,
  confidence = 0,
  featuresUsed = {},
  missingOptional = [],
  missingCritical = [],
  leakageSafe = true,
  version = '1.0.0',
  ...rest
}) {
  return {
    market,
    status,
    fairProb,
    fairLine,
    confidence,
    featuresUsed,
    missingOptional,
    missingCritical,
    diagnostics: {
      leakageSafe,
      version,
    },
    ...rest,  // sport-specific fields preserved
  };
}

module.exports = { buildModelOutput };
```

## Ablation Test Pattern

```js
// ablation.test.js

describe('MLB F5 Total — ablation tests', () => {
  const baseCase = makeMlbF5Fixture();  // full valid inputs

  test('starter_skill_ra9 null → NO_BET', () => {
    const ablated = { ...baseCase, homePitcher: { ...baseCase.homePitcher, siera: null, xfip: null, xera: null } };
    const result = projectF5TotalCard(ablated.homePitcher, ablated.awayPitcher, ablated.f5Line, ablated.context);
    expect(result.status).toBe('NO_BET');
  });

  test('wrc_plus null → NO_BET', () => {
    const ablated = { ...baseCase, context: { ...baseCase.context, home_offense_profile: { ...baseCase.context.home_offense_profile, wrc_plus: null, wrc_plus_vs_rhp: null, wrc_plus_vs_lhp: null } } };
    const result = projectF5TotalCard(ablated.homePitcher, ablated.awayPitcher, ablated.f5Line, ablated.context);
    expect(result.status).toBe('NO_BET');
  });

  test('removing park factor degrades projection by >= 10%', () => {
    const full = projectF5Total(baseCase.homePitcher, baseCase.awayPitcher, baseCase.context);
    const ablated = projectF5Total(baseCase.homePitcher, baseCase.awayPitcher, { ...baseCase.context, park_run_factor: null });
    expect(ablated.status).toBe('NO_BET');  // park_run_factor is required
  });
});

describe('NHL Total — ablation tests', () => {
  test('goalie save pct null → confidence drops by >= 0.05', () => {
    const full = predictNHLGame(makeNhlFixture());
    const ablated = predictNHLGame({ ...makeNhlFixture(), homeGoalieSavePct: null, awayGoalieSavePct: null });
    expect(ablated.confidence).toBeLessThan(full.confidence - 0.04);
  });

  test('double unknown goalie → NO_BET via cross-market gate', () => {
    // Tested via WI-0820; this ablation verifies from the pace model side
    const result = predictNHLGame({ ...makeNhlFixture(), homeGoalieCertainty: 'UNKNOWN', awayGoalieCertainty: 'UNKNOWN' });
    expect(result.goalieConfidenceCapped).toBe(true);
    expect(result.confidence).toBeLessThanOrEqual(0.35);
  });
});

describe('NBA Canonical — ablation tests', () => {
  test('pace null → returns null (NO_BET after WI-0820)', () => {
    const result = projectNBACanonical(115, 110, null, 112, 108, null);
    expect(result).toBeNull();
  });

  test('offense rating null → returns null', () => {
    const result = projectNBACanonical(null, 110, 103, 112, 108, 101);
    expect(result).toBeNull();
  });
});
```
